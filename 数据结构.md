二叉树

```java
public int getIndex(@RequestParam("arr") int[] arr,@RequestParam("val") int val)
{
        int min=0;
        int max=arr.length-1;
        int mid=(min+max)/2;
        while (arr[mid]!=val){
            if(arr[mid]<val){
                min=min+1;
            }
            if(arr[mid]>val){
                max=mid-1;
            }
            if(min>max){
                return -1;
            }
            mid=(min+max)/2;
        }
        return mid;
}
```
